16  
17      public void run() {
18          ui.run();
19          
20          Difficulty difficulty = ui.selectDifficulty();
21  
22          while (series.hasNext()) {
23              Question question = series.next();
24              String guess = ui.askQuestion(question, difficulty);
25              
26              ui.consequence(question.isCorrect(guess));
27          }
28      }
29  
30      private void setSeries() {
31          QuestionReader reader = new QuestionReader(null);
32          reader.readQuestions(series);
33      }
