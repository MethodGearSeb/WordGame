3   import com.mgseb.wordgame.domain.Question;
4   import java.io.File;
5   import java.io.FileNotFoundException;
6   import java.util.Scanner;
7   
8   public class QuestionReader {
9   
10      private final String fileAddress;
11  
12      /**
13       * If file address given is null, use default file address.
14       *
15       * @param fileAddress
16       */
17      public QuestionReader(String fileAddress) {
18          this.fileAddress = replaceIfInvalid(fileAddress);
19      }
20  
21      public void readQuestions(QuestionSeries series) {
22          Scanner scanner = openScanner();
23  
24          if (scanner == null) {
25              return;
26          }
27  
28          while (scanner.hasNext()) {
29              String line = scanner.nextLine();
30              String hint = "";
31              String answer = "";
32  
33              if (line.contains("@")) {
34                  hint = "";
35                  line = scanner.nextLine();
36  
37                  while (!line.contains("#")) {
38                      hint += line;
39                      line = scanner.nextLine();
40                  }
41              }
42  
43              if (line.contains("#")) {
44                  line = scanner.nextLine();
45                  answer = line;
46              }
47  
48              Question question = new Question(hint, answer);
49              series.addQuestion(question);
50          }
51      }
52  
53      private String replaceIfInvalid(String fileAddress) {
54          if (fileAddress != null && !fileAddress.isEmpty()) {
55              return fileAddress;
56          }
57          return "wordgame/english.txt";
58      }
59  
60      private Scanner openScanner() {
61          ClassLoader loader = getClass().getClassLoader();
62          File file = new File(loader.getResource(fileAddress).getFile());
63          Scanner scanner = null;
64          try {
65              scanner = new Scanner(file);
66          } catch (FileNotFoundException e) {
67              System.out.println("ERROR: File could not be found");
68          }
69          return scanner;
70      }
71  }
