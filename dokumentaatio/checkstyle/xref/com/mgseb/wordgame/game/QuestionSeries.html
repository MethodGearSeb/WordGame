1   package com.mgseb.wordgame.game;
2   
3   import com.mgseb.wordgame.domain.Question;
4   import java.util.ArrayList;
5   
6   /**
7    * Collection of questions.
8    *
9    * @author fuksi
10   */
11  public class QuestionSeries {
12  
13      private final ArrayList<Question> questionList;
14  
15      /**
16       * Initialises attribute.
17       */
18      public QuestionSeries() {
19          this.questionList = new ArrayList<>();
20      }
21  
22      /**
23       * Adds question to the collection.
24       *
25       * @param question to be added
26       */
27      public void addQuestion(Question question) {
28          questionList.add(question);
29      }
30  
31      /**
32       * Picks a random question from the list, deletes it, and returns it.
33       *
34       * @return a question for examination
35       */
36      public Question next() {
37          if (questionList.isEmpty()) {
38              return null;
39          }
40  
41          int i = (int) Math.floor(Math.random() * questionList.size());
42          Question question = questionList.get(i);
43          questionList.remove(question);
44          return question;
45      }
46  
47      /**
48       * Checks to see that the collection is not depleted. Meant to prevent
49       * problems derived from the absence of questions.
50       *
51       * @return the information of whether there are questions left
52       */
53      public boolean hasNext() {
54          return questionList.size() > 0;
55      }
56  }
