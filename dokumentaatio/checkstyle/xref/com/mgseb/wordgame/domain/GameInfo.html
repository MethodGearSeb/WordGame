1   package com.mgseb.wordgame.domain;
2   
3   import java.awt.Color;
4   
5   /**
6    * Keeps track of game progress and score. Manages messages that are displayed
7    * to the user.
8    *
9    * @author fuksi
10   */
11  public class GameInfo {
12  
13      private String message;
14      private Color messageColor;
15      private int questionNumber;
16      private int right;
17      private int wrong;
18  
19      /**
20       * Initialises most class attributes.
21       */
22      public GameInfo() {
23          message = "Get started!";
24          questionNumber = 1;
25          right = 0;
26          wrong = 0;
27      }
28  
29      /**
30       * Updates game progress information. Called anytime a question is answered.
31       * Updates question number, score, message, message colour.
32       *
33       * @param correct tells whether the latest guess made by the user was right
34       */
35      public void nextQuestion(boolean correct) {
36          questionNumber++;
37  
38          if (correct) {
39              right++;
40              message = "You the best!";
41              messageColor = Color.GREEN;
42          } else {
43              wrong++;
44              message = "You the worst!";
45              messageColor = Color.RED;
46          }
47      }
48  
49      public String getMessage() {
50          return message;
51      }
52  
53      public Color getMessageColor() {
54          return messageColor;
55      }
56  
57      public int getQuestionNumber() {
58          return questionNumber;
59      }
60  
61      public int getRight() {
62          return right;
63      }
64  
65      public int getWrong() {
66          return wrong;
67      }
68  
69      /**
70       * Reverts the attributes related to the message back to neutral.
71       */
72      public void defaultMessage() {
73          message = "Guess the word.";
74          messageColor = Color.BLACK;
75      }
76  }
